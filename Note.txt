Deploy============
# 1. Clone source
sudo apt-get update -y && sudo apt-get install -y git
sudo git clone https://github.com/manhdauvn09-manhds/okr-web.git /opt/okr-web
sudo chown -R $USER:$USER /opt/okr-web
cd /opt/okr-web

# 2. Cấp quyền & chạy deploy
chmod +x deploy.sh
./deploy.sh
